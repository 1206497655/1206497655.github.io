dojo学习笔记--创建类的工具declare
dojo declare
2018-07-29

  declare(className,superclass,props) 是创建类的基础，es6之前js并不支持创建类，没有关键词class和extends.declare允许开发者实现多继承，从而创建灵活的有弹性的代码，避免重复写代码。
@param: className
  类型: String
  描述: 可选。className参数是要创建的这个类的名字，也包括它的命名空间。
    命名类（给定了className, 相对的为匿名类， Named Class）的名字
    会存在于全局作用域， className也可以通过命名空间来表示继承链。
    当省略参数时，创建的类为匿名类，需要使用变量接收。匿名类只在给
    定的作用域有效。
@param: superclass
  类型: Function | Function[]
  描述: SuperClass 参数可以为null(但不能省略)、 单个已存在的类或者多个
    类组成的数组。
@param: props
  类型: Object
  描述: 包含这个类原型方法和原型属性的一个对象。 此参数提供的属性或方法
    如果跟被继承的类的属性或方法同名，那么将会被优先使用。

require([
            'dojo/_base/declare',
            'dojo/_base/lang',
            'dojo/dom',
            "dijit/form/Button",
            "dojo/domReady!"
        ], function (declare, lang, dom, Button) {
            // className(类名)
            // 命名类
            declare("mynamespace.Person", null, {
                // Custom properties and methods here
                method: function () {
                    console.log("命名类");
                }
            })
            var per = new mynamespace.Person();
            per.method();
            // 匿名类  需要变量名去接收
            var Person = declare(null, {
                // Custom properties and methods here
                method: function () {
                    console.log("匿名类");
                }
            })
            var per = new Person();
            per.method();

            // SuperClass(超类)，继承的优先级从左至右依次升高，若果出现重复属性或方法时，
            // 时，优先级高的将会被保留，低的将会被覆盖。可以通过declare()的第三个参数添加
            // 相同的属性或方法名来覆盖你想覆盖掉的东西。
            //不继承
            console.log("------------不继承-----------")
            declare("myClass", null, {
                text: "不继承"
            });
            var myInst = new myClass();
            console.log(myInst);
            console.log(myInst.text);

            //单继承
            console.log("------------单继承-----------")
            declare("mySubClass", myClass, {
                text: "单继承"
            });
            var myInst = new mySubClass();
            console.log(myInst);
            console.log(myInst.text)

            // 多继承
            console.log("------------多继承-----------")
            declare("myFirstClass", null, {
                num1: "1",
                prop: "a"
            });
            declare("mySecondClass", null, {
                num2: "2",
                prop: "b"
            });
            declare("myThirdClass", null, {
                num3: "3",
                prop: "c"
            });
            var Obj = {
                num4: "4",
                prop: "d"
            }
            declare("myMultiSubClass", [myFirstClass, mySecondClass, myThirdClass], Obj);
            var myInst = new myMultiSubClass();
            console.log(myInst);
            console.log(myInst.num1 + myInst.num2 + myInst.num3 + myInst.num4 + "," + myInst.prop);
            /* 
                从myInst打印的情况可以看出，优先级的具体表现就是原型链继承的先后问题，Obj处于
                原型链的首位，所以优先级最高，原型链依次往后分别是myThirdClass、mySecondClass、
                myFirstClass，前面提到出现重复项时优先级高的class将会覆盖掉优先级低的，其实
                严格来讲并不是这样，根据原型链的特性，当检索一个对象实例的属性时，会首先检查实
                例自己本身的属性，如果没有，会查找按顺序原型链，如果原型链中某一个对象出现有这个属性，
                则直接返回这个属性的值，不再进行查找，如果整个原型链都没有这个属性则返回undefine。
                也正是因为这样，多继承类出现重复定义项时，排列在原型链前的将会被直接返回，因为
                已经返回，所以永远得不到原型链后面的重复项的值，这个并不是覆盖，重复项还是处于
                原型链中，只是得不到而已。
                我们可以做个实验直接获取myFirstClass的prop属性，查看它是否还存在以及属性值是
                否被改变。
            */
            console.log(myInst.__proto__.__proto__.__proto__.__proto__.prop)//myFirstClass中的prop属性值仍然为"a"


            // 属性和方法对象
            // declare方法的最后的参数是包含这个类原型方法和原型属性的一个对象。 此参数提供的
            // 属性或方法如果跟被继承的类的属性或方法同名，那么将会被优先使用。
            // 拥有属性和方法的类
            declare("MyClass", null, {
                myProperty1: 12,
                myOtherProperty: "Hello",
                myMethod: function () {
                    console.log("World");
                    return result;
                }
            });

            //类的创建和继承例子
            //使用declare()继承dijit/form/Button组件，并添加label和onClick属性。
            declare("newButton", Button, {
                label: "My Button",
                onClick: function (e) {
                    console.log("I was clicked!");
                }
            })
            var myButton = new newButton({}, "declare").startup();

            // 构造函数 constructor
            // 类中特别的一个方法就是constructor方法，constructor方法会在类的实例化时触发，在
            // 新对象的作用域名内执行。如果在constructor中使用this，则这个this则会指向这个类的
            // 实例，而不是类。constructor接受实例化指定的参数。
            console.log("--------------构造函数-------------")
            declare("Twitter", null, {
                username: "defaultUser",

                constructor: function (args) {
                    declare.safeMixin(this, args);//将传入的参数混入到实例对象中
                }
            })
            var myInst = new Twitter();
            console.log(myInst.username);//输出defaultUser
            //创建并传入一个对象
            var myInst = new Twitter({
                username: "Sparrow"
            });
            console.log(myInst.username);//输出Sparrow

            // 如果继承的原型中属性指向Object或者Array，创建的实例将会指向同一个值。下面来探讨一
            // 种数据被意外修改的情况
            var arr = [1, 2, 3, 4];
            declare("myClass", null, {
                num: arr,
                prop: "aaa"
            });
            var myInst1 = new myClass();
            var myInst2 = new myClass();

            myInst1.prop = "bbb"; //修改num属性
            console.log(myInst1.prop); // bbb
            console.log(myInst2.prop); // aaa
            // 此时实际上时给myInst新增了一个prop属性，并没有操作原型链中的prop属性，所以myInst2.prop并没有被修改

            myInst1.__proto__.prop = "bbb";
            console.log(myInst1.prop); // bbb
            console.log(myInst2.prop); // bbb
            // 操作原型链时，其他实例也会受到影响，这种情况一般不会发生

            myInst1.num = [4, 5, 6, 7];
            console.log(myInst1.num); // [4,5,6,7]
            console.log(myInst2.num); // [1,2,3,4]
            // 此时实际上时给myInst新增了一个num属性，并没有操作原型链中的num属性，所以myInst2.num并没有被修改

            delete myInst1.num;//删除num属性，以下的myInst1已经没有这个属性

            myInst1.num.push(5);
            console.log(myInst1.num); // [1,2,3,4,5]
            console.log(myInst2.num); // [1,2,3,4,5]
            // push一个元素进myInst1.num时，myInst2.num也受到了影响，因为实例本身没有这个属性，
            // 所以push操作的实际上是原型链中的num，为了避免这种非故意的修改共享数据，对象的属性
            // 声明为null，然后在构造函数constructor中对它进行初始化。当然这是网上的一种说法，经
            // 过实践得出这说法不完全可靠。
            console.log("-------------使用构造函数改造--------------")
            //情况1
            var arr2 = [1, 2, 3, 4];
            declare("myClass", null, {
                numArr: null,
                prop: null,
                constructor: function (arr) {
                    this.numArr = arr;
                    this.prop = "aaa";
                }
            });
            var myInst1 = new myClass([1, 2, 3, 4]);
            var myInst2 = new myClass([1, 2, 3, 4]);
            //push一个5进myInst1.num
            myInst1.numArr.push(5);
            console.log(myInst1.numArr); // [1, 2, 3, 4, 5]
            console.log(myInst2.numArr); // [1, 2, 3, 4]
            // 如果使用传参的方式进行初始化，并且是匿名数组，那么改变myInst1时并不会对myInst2造成影响

            var myInst1 = new myClass(arr2);
            var myInst2 = new myClass(arr2);
            //push一个5进myInst1.num
            myInst1.numArr.push(5);
            console.log(myInst1.numArr); // [1, 2, 3, 4, 5]
            console.log(myInst2.numArr); // [1, 2, 3, 4，5]
            // 如果使用传递的参数不是匿名数组，那么同样会出现误改数据的情况
            // 原因很简单，无论怎样初始化，数组类型的引用特性是不会改变的，"this.numArr = arr2"只是起
            // 到一个把numArr指向arr2的作用，无论this的对象是谁。可以了解一下浅拷贝和引用类型的特性
            // 便可知道。若果要使用共享的数据而不希望出现误改，根本的解决办法是使用深度拷贝，把arr2的值
            // 完完整整的拷贝过来而不是使用指针。但这样的一个缺点就是当arr或者obj占用空间很大时，使用
            // 深度拷贝之后的空间会翻倍，所以应该谨慎使用，或者使用完之后应该及时销毁。
            declare("myClass", null, {
                numArr: null,
                prop: null,
                constructor: function (arr) {
                    this.numArr = lang.clone(arr);//clone()是dojo/_base/lang的一个深度拷贝工具。
                    this.prop = "aaa";
                }
            });
            var myInst1 = new myClass(arr2);
            var myInst2 = new myClass(arr2);
            //push一个5进myInst1.num
            myInst1.numArr.push(5);
            console.log(myInst1.numArr); // [1, 2, 3, 4, 5]
            console.log(myInst2.numArr); // [1, 2, 3, 4]

            // this.inherited() 虽然重写一个方法确实有用， 但每一个类的的构造函数都应该执行它继承链中的
            // 父类构造函数，以保持它原始的功能。 这样this.inherited(arguments)语句就派上用场了。  
            // this.inherited(arguments)语句调用父类的同名方法。 
            var A = declare(null, {
                myMethod: function () {
                    console.log("Hello!");
                }
            });
            var B = declare(A, {
                myMethod: function () {
                    // 调用父类A中同名的方法
                    this.inherited(arguments); // arguments指的就是在父类A中定义的myMethod
                    console.log("World!");
                }
            });
            var myB = new B();
            myB.myMethod();
            //      Hello!
            //      World!
        })
