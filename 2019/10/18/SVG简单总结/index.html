简单总结

1. 回顾知识点

简单学习了SVG图形设计语言的部分知识点，包括：

  • 基础图形
  • 坐标系统
  • 颜色渐变和图案
  • 最常见的Path元素
  • 文本
  • 图形引用裁剪和蒙版
  • 动画

SVG大大小小共有150+个元素，想要全部掌握需要一定的时间成本，但是掌握了前面我们学习过的内容，就能够大致地让我们知道SVG的运作原理以及基本用法，遇到使用SVG技术做渲染的场景就会比较容易理解，也能编写简易的可视化表达图形，下面来应用我们学习过的知识来编写一些综合实例。

2. 应用例子

2.1 loading状态图标

loading状态可以说是非常常见的需求，带有动画效果的loading状态通常通过gif图片、icon+CSS3或者SVG图标实现。gif图片的缺点是显而易见的，会增加网络请求。icon+CSS3虽然没有网络请求，但是只能是单色图标。而SVG就没有这些缺点了，甚至还可以通过代码自己设计想要的图标模样。下面是一个简单的多色旋转图标，用到的技术是SVG基础图形+渐变效果+动画，如下:


<svg xmlns="http://www.w3.org/2000/svg" width="50" height="50" viewBox="-25 -25 50 50">
  <defs>
    <linearGradient id="grad" x1="0" y1="0" x2="1" y2="1">
        <stop class="stop1" offset="0" stop-color="#1497fc"/>
        <stop class="stop2" offset="0.5" stop-color="#a469be"/>
        <stop class="stop3" offset="1" stop-color="#ff8c00"/>
    </linearGradient>
  </defs>
  <circle cx="0" cy="0" r="15" stroke="url(#grad)" stroke-width="6" fill="none">
    <animateTransform attributeName="transform" begin="0s" dur="1s" type="rotate" from="0 0 0" to="360 0 0" repeatCount="indefinite"/>
  </circle>
</svg>

2.2 饼图

饼图是最常见的图表类组件之一，通过SVG技术我们也可以设计出自己想要的饼图，下面是用SVG Path元素绘制的一个简单饼图，关键点在于如何计算扇区弧线的末端点：

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>pie</title>
  <style>
    .pie-container {
      position: relative;
      width: 400px;
      height: 400px;
    }
  </style>
</head>

<body>
  <div class="pie-container" id="pieContainer">
  </div>
  <script>
    const Pie = function (node) {
      this.node = this.init(node)
    }
    Pie.prototype = {
      // 初始化饼图
      init: function (node) {
        let root = null
        if (typeof (node) === 'string') {
          root = document.getElementById(node)
        } else if (typeof (node) === 'object' && !Array.isArray(node)) {
          root = node
        } else {
          return false
        }
        const SVG = document.createElementNS('http://www.w3.org/2000/svg', 'svg')
        SVG.setAttribute('width', '100%')
        SVG.setAttribute('height', '100%')
        SVG.setAttribute('viewBox', '-120 -120 240 240')
        SVG.setAttribute('transform', 'rotate(-90)')
        root.appendChild(SVG)
        return SVG
      },
      // 渲染器
      render: function (percents, point, i) {
        const path = document.createElementNS('http://www.w3.org/2000/svg', 'path')
        let beginPoint = null // 圆弧起点
        let endPoint = null // 圆弧结束点
        let x0 = 0 // 起始点x
        let y0 = 0 // 起始点y
        let a = 0 // 圆心角
        const r = 100 // 半径
        x0 = point[0] || 100
        y0 = point[1] || 0
        // 计算偏移角度
        a = (function (ps) {
          let percent = 0
          ps.forEach((p, j) => {
            if (j <= i) {
              percent += p
            }
          })
          return percent * 360
        })(percents)
        // 已知半径、圆心角，求圆弧的末端坐标点
        let x1 = r * Math.cos(a * Math.PI / 180)
        let y1 = r * Math.sin(a * Math.PI / 180)
        beginPoint = `${x0},${y0}`
        endPoint = `${x1},${y1}`
        path.setAttribute('d', `M0,0 L${beginPoint}A100,100,0,0,1,${endPoint}Z`)
        path.setAttribute('fill', `hsl(${360 * i / datas.length} ,70%,60%)`)
        return { path, point: [x1, y1] }
      },
      // 加载数据
      setData: function (datas) {
        if (!Array.isArray(datas)) {
          return
        }
        // 整理出基本的信息
        const option = {
          total: 0, // 总数
          keys: [], // 条目名称
          percents: [] // 百分比
        }
        datas.forEach(data => {
          option.total += Object.values(data)[0]
          option.keys.push(Object.keys(data)[0])
        })
        let lastPercent = 1
        datas.forEach((data, i) => {
          if (i < datas.length - 1) {
            const percent = parseFloat((Object.values(data)[0] / option.total).toFixed(2))
            option.percents.push(percent)
            lastPercent -= parseFloat(percent.toFixed(2))
          } else {
            option.percents.push(lastPercent)
          }
        })
        // 通过path元素来绘制每个扇区
        let lastEndPoint = [] // 记录上一个endPonint作为下一个扇形圆弧的起点
        datas.forEach((data, i) => {
          const res = this.render(option.percents, lastEndPoint, i)
          this.node.appendChild(res.path)
          lastEndPoint = res.point
        })
        return option
      },
      destory: function () {
        this.node.parent.removeChild(this.node)
        this.node = null
      }
    }

    const datas = [
      { '鱼': 50 },
      { '虾': 60 },
      { '蟹': 70 }
    ]
    const pie = new Pie('pieContainer')
    pie.setData(datas, pie)
  </script>
</body>

</html>

2.3 力导向图

力导向图通常用来表示关系。下面是一个每个点之间具有两两关联关系的力导向图，每个点的位置随机给定，然后通过每秒60次的频率更新节点位置，经过一定时间的平衡后呈现位一个规则多边形的图形（SVG的SMIL动画在此并不适用，本例使用脚本来实现动画效果）。本例的关键在于如何更新节点位置使其动画效果较好并且能够达到平衡规则状态，答案便是弹簧模型。物体在被多个弹簧牵引初期，会处于一个晃动的状态，随着时间的推移动能逐渐磨损而达到一个平衡静止的状态。本例算法的实现可能比较复杂。

弹簧模型:

  • 物体受弹簧力公式: Fi = k·x 其中k表示弹簧系数，x表示弹簧长度变化的值。
  • 两物体之间的合力: F = ∑Fi 两力之和
  • 加速度: a = F/m m表示物体质量
  • 速度: v = v0+a·Δt
  • 位移: s = s0+v·Δt

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Graph</title>
  <style>
    .graph-container .refresh {
      position: absolute;
      right: 0;
      top: 0;
    }

    .graph-container {
      position: relative;
      width: 600px;
      height: 600px;
    }
  </style>
</head>

<body>

  <div class="graph-container" id="graphContainer">
    <button class="refresh" onclick="reset()">重置</button>
  </div>
  <script>
    // 力的计算模型
    const Vector = function (x, y) {
      this.x = x || 0
      this.y = y || 0
      this.k = 5 // 弹性系数
      this.n = 0.98 // 能量损耗系数
      this.l = 300 // 表示弹簧的长度
    }
    Vector.fromPoints = function (p1, p2) {
      return new Vector(p2.x - p1.x, p2.y - p1.y)
    }
    Vector.prototype = {
      constructor: Vector,
      square: function () {
        return this.x * this.x + this.y * this.y
      },
      root: function () {
        return Math.sqrt(this.square())
      },
      length: function () {
        return Math.sqrt(this.square())
      },
      add: function (q) {
        return new Vector(this.x + q.x, this.y + q.y)
      },
      minus: function (q) {
        return new Vector(this.x - q.x, this.y - q.y)
      },
      multipy: function (scale) {
        return new Vector(this.x * scale, this.y * scale)
      },
      normalize: function (length) {
        if (length === undefined) {
          length = 1
        }
        return this.multipy(length / this.length())
      }
    }

    // 力导向图
    const Graph = function (node) {
      this.points = []
      this.node = this.init(node)
      this.animation = null
    }
    Graph.prototype = {
      // 初始化节点
      init: function (node) {
        let root = null
        if (typeof (node) === 'string') {
          root = document.getElementById(node)
        } else if (typeof (node) === 'object' && !Array.isArray(node)) {
          root = node
        } else {
          return false
        }
        const SVG = document.createElementNS('http://www.w3.org/2000/svg', 'svg')
        SVG.setAttribute('width', '100%')
        SVG.setAttribute('height', '100%')
        SVG.setAttribute('viewBox', '-300 -300 600 600')
        root.appendChild(SVG)
        return SVG
      },
      // 随机数
      random: function (min, max) {
        return Math.round(min + (max - min) * Math.random())
      },
      // 圆点
      circle: function (x, y, r, color, parent) {
        const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle')
        circle.mousedown = false
        circle.setAttribute('cx', x)
        circle.setAttribute('cy', y)
        circle.setAttribute('r', r)
        circle.setAttribute('fill', color)
        circle.setAttribute('style', 'cursor: pointer;')
        return circle
      },
      // 渲染器
      render: function (data) {
        if (!Array.isArray(data)) return
        const path = document.createElementNS('http://www.w3.org/2000/svg', 'path')
        path.setAttribute('stroke', 'gray')
        this.node.appendChild(path)
        data.forEach((d, i) => {
          const point = {}
          const x = this.random(-300, 300)
          const y = this.random(-300, 300)
          const r = this.random(10, 20)
          const color = `hsl(${360 * i / data.length}, 100%, 60%)`
          const circle = this.circle(x, y, r, color, this.node)
          point.node = circle // 节点
          point.s = new Vector(x, y) // 点的位置
          point.v = new Vector() // 点的速度
          point.a = new Vector() // 点的加速度
          point.m = r / 10 // 点的质量
          this.node.appendChild(circle)
          this.points.push(point)
        })
        let link = []
        this.points.forEach(pa => {
          this.points.forEach(pb => {
            link = link.concat([
              'M', pa.s.x, pa.s.y,
              'L', pb.s.x, pb.s.y
            ])
          })
        })
        const d = link.join(' ')
        path.setAttribute('d', d)
        let lastFrameTime = new Date()
        this.update(this.points, lastFrameTime, path)
      },
      // 更新
      update: function (points, lastFrameTime, path) {
        const frameTime = new Date()
        let t = (frameTime - lastFrameTime) / 1000 // 每一帧的时间, 单位秒
        let isMoving = false // 移动中
        //点位置更新
        points.forEach(function (pa) {
          let f = new Vector()
          //计算合力
          points.forEach(function (pb) {
            if (pa == pb) return
            const x = Vector.fromPoints(pa.s, pb.s)
            const delta = x.length() - f.l
            //f = k * x
            f = f.add(x.normalize(delta * f.k))
          })
          pa.a = f.multipy(pa.m)
          pa.v = pa.v.add(pa.a.multipy(t)).multipy(f.n)
          // 任意点速度大于0.01表示平衡未完成
          if (pa.v.root() > 0.1 || pa.v.root() === 0) {
            isMoving = true
          }
          pa.s = pa.s.add(pa.v.multipy(t))
          pa.node.setAttribute('cx', pa.s.x)
          pa.node.setAttribute('cy', pa.s.y)
        })
        if (!isMoving) {
          return
        }
        //连线更新
        let linkPath = []
        points.forEach(function (pa) {
          var sa = pa.s
          points.forEach(function (pb) {
            if (pa == pb) return
            var sb = pb.s
            linkPath = linkPath.concat([
              'M', pa.s.x, pa.s.y,
              'L', pb.s.x, pb.s.y
            ])
          })
        })
        path.setAttribute('d', linkPath.join(' '))
        this.animation = window.requestAnimationFrame(this.update.bind(this, this.points, frameTime, path))
      },
      // 加载数据
      setData: function (data) {
        this.render(data)
      },
      // 销毁
      destory: function () {
        window.cancelAnimationFrame(this.animation)
        this.node.parentNode.removeChild(this.node)
        this.points = null
        this.node = null
        this.prototype = null
        this.animation = null
      }
    }

    // 数据
    const data = 'abcdefghijk'.split('')
    // const data = 'ab'.split('')
    let graph = new Graph('graphContainer')
    graph.setData(data)

    // 重置
    function reset() {
      graph.destory()
      graph = new Graph('graphContainer')
      graph.setData(data)
    }
  </script>
</body>

</html>
